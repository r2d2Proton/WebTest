        <div class="status-line"><span class="status-label">Status:</span> <span id="status" class="status-loading">Initializing...</span></div>
        <div class="status-line"><span class="status-label">Scene:</span> <span id="sceneStatus">-</span></div>
        <div class="status-line"><span class="status-label">Physics Engine:</span> <span id="physicsStatus" class="status-loading">Loading...</span></div>
        <div class="status-line"><span class="status-label">Scene Data:</span> <span id="sceneDataStatus">-</span></div>
        <div class="status-line"><span class="status-label">Model:</span> <span id="modelStatus">-</span></div>
        <div class="status-line"><span class="status-label">Render Loop:</span> <span id="renderStatus">-</span></div>
        <div class="status-line"><span class="status-label">Object State:</span> <span id="objectState">-</span></div>
        <div class="status-line" id="errorLine" style="display:none;"><span class="status-label">Error:</span> <span id="errorMsg" class="status-error">-</span></div>
    </div>

    <script>
        let engine;
        let scene;
        let ground;
        let currentModel = null;
        let sceneFiles = [];

        window.play = function() {
            try {
                if (currentModel) {
                    scene.animationGroups.forEach(animationGroup => {
                        animationGroup.play(true);
                    });
                    updateStatus('objectState', 'Playing', 'status-success');
                    console.log('Play called');
                } else {
                    console.warn('Model not available');
                }
            } catch (error) {
                console.error('Play error:', error);
            }
        };

        window.pause = function() {
            try {
                if (currentModel) {
                    scene.animationGroups.forEach(animationGroup => {
                        animationGroup.pause();
                    });
                    updateStatus('objectState', 'Paused', 'status-loading');
                    console.log('Pause called');
                }
            } catch (error) {
                console.error('Pause error:', error);
            }
        };

        window.stop = function() {
            try {
                if (currentModel) {
                    scene.animationGroups.forEach(animationGroup => {
                        animationGroup.stop();
                    });
                    updateStatus('objectState', 'Stopped', 'status-error');
                    console.log('Stop called');
                }
            } catch (error) {
                console.error('Stop error:', error);
            }
        };

        function updateStatus(id, message, className = '') {
            const element = document.getElementById(id);
            if (element) {
                element.textContent = message;
                if (className) {
                    element.className = className;
                }
            }
        }

        function showError(message) {
            document.getElementById('errorLine').style.display = 'block';
            updateStatus('errorMsg', message, 'status-error');
            updateStatus('status', 'Error', 'status-error');
        }

        async function fetchSceneFiles() {
            try {
                updateStatus('status', 'Fetching file list...', 'status-loading');
                
                const response = await fetch('https://api.github.com/repos/r2d2Proton/PublicSamples/contents/Assets/Scenes');
                const files = await response.json();
                
                sceneFiles = files
                    .filter(f => f.name.endsWith('.glb'))
                    .map(f => ({
                        name: f.name.replace('.glb', '').replace(/_/g, ' '),
                        url: f.download_url
                    }));

                const select = document.getElementById('sceneSelect');
                select.innerHTML = '';
                sceneFiles.forEach((file, index) => {
                    const option = document.createElement('option');
                    option.value = index;
                    option.textContent = file.name;
                    select.appendChild(option);
                });

                if (sceneFiles.length > 0) {
                    select.selectedIndex = 0;
                    updateStatus('status', 'Ready to load', 'status-success');
                } else {
                    updateStatus('status', 'No scenes found', 'status-error');
                }
            } catch (error) {
                showError('Failed to fetch scene files: ' + error.message);
            }
        }

        async function loadSceneModel(fileIndex) {
            try {
                if (fileIndex < 0 || fileIndex >= sceneFiles.length) return;

                const file = sceneFiles[fileIndex];
                updateStatus('sceneDataStatus', `Loading ${file.name}...`, 'status-loading');

                if (currentModel) {
                    currentModel.dispose();
                    currentModel = null;
                }

                const result = await BABYLON.SceneLoader.ImportMeshAsync("", file.url, "", scene);
                
                const root = result.meshes[0];
                if (root) {
                    currentModel = root;
                    
                    const trParent = new BABYLON.TransformNode("modelParent", scene);
                    root.parent = trParent;
                    root.scaling.scaleInPlace(100);
                    root.position.y = 4;

                    result.meshes.forEach(mesh => {
                        if (mesh !== root && !mesh.name.includes("__")) {
                            try {
                                const aggregate = new BABYLON.PhysicsAggregate(
                                    mesh,
                                    BABYLON.PhysicsShapeType.CONVEX_HULL,
                                    { mass: 1, restitution: 0.75 },
                                    scene
                                );
                            } catch (e) {
                                console.warn(`Could not create physics for mesh ${mesh.name}:`, e);
                            }
                        }
                    });
                }

                updateStatus('modelStatus', file.name, 'status-success');
                updateStatus('objectState', 'Loaded', 'status-success');
                updateStatus('status', 'Ready', 'status-success');
            } catch (error) {
                showError('Error loading model: ' + error.message);
            }
        }

        async function loadHavokScript() {
            return new Promise((resolve, reject) => {
                const response = fetch('https://cdn.jsdelivr.net/gh/r2d2Proton/PublicSamples@19d8bdbb78f9c1f55cc7dcd07ae04b95e5f9eb47/Assets/Physics/HavokPhysics_umd.js')
                    .then(r => r.text())
                    .then(scriptText => {
                        // Execute the script in global context
                        eval(scriptText);
                        
                        // Wait for HavokPhysics to be available
                        let attempts = 0;
                        const checkHavok = setInterval(() => {
                            if (typeof HavokPhysics !== 'undefined') {
                                clearInterval(checkHavok);
                                resolve();
                            } else if (attempts > 50) {
                                clearInterval(checkHavok);
                                reject(new Error('HavokPhysics failed to initialize'));
                            }
                            attempts++;
                        }, 100);
                    })
                    .catch(reject);
            });
        }

        (async function() {
            try {
                updateStatus('status', 'Creating scene', 'status-loading');
                
                const canvas = document.getElementById("renderCanvas");
                engine = new BABYLON.Engine(canvas, true);
                scene = new BABYLON.Scene(engine);
                updateStatus('sceneStatus', 'Ready', 'status-success');

                const camera = new BABYLON.FreeCamera("camera1", new BABYLON.Vector3(0, 5, 10), scene);
                camera.setTarget(BABYLON.Vector3.Zero());
                camera.attachControl(canvas, true);

                const light = new BABYLON.HemisphericLight("light1", new BABYLON.Vector3(0, 1, 0), scene);
                light.intensity = 0.7;

                ground = BABYLON.MeshBuilder.CreateGround("ground", {width: 10, height: 10}, scene);

                updateStatus('physicsStatus', 'Loading Havok...', 'status-loading');
                
                // Load Havok script using jsDelivr CDN (better for GitHub files)
                await loadHavokScript();
                
                updateStatus('physicsStatus', 'Initializing...', 'status-loading');
                
                // Initialize HavokPhysics with WASM from jsDelivr
                const havokInstance = await HavokPhysics({
                    locateFile: (file) => `https://cdn.jsdelivr.net/gh/r2d2Proton/PublicSamples@19d8bdbb78f9c1f55cc7dcd07ae04b95e5f9eb47/Assets/Physics/${file}`
                });
                
                const havokPlugin = new BABYLON.HavokPlugin(true, havokInstance);
                scene.enablePhysics(new BABYLON.Vector3(0, -9.8, 0), havokPlugin);
                
                updateStatus('physicsStatus', 'Ready', 'status-success');
                
                const groundAggregate = new BABYLON.PhysicsAggregate(
                    ground, 
                    BABYLON.PhysicsShapeType.BOX, 
                    { mass: 0, restitution: 0.5 }, 
                    scene
                );
                
                try {
                    const hdrTexture = BABYLON.CubeTexture.CreateFromPrefilteredData("textures/environment.dds", scene);
                    const currentSkybox = scene.createDefaultSkybox(hdrTexture, true);
                } catch (e) {
                    console.warn('Skybox texture not available:', e);
                }

                await fetchSceneFiles();

                if (sceneFiles.length > 0) {
                    await loadSceneModel(0);
                }

                document.getElementById('sceneSelect').addEventListener('change', (e) => {
                    loadSceneModel(parseInt(e.target.value));
                });

                updateStatus('renderStatus', 'Starting...', 'status-loading');
                
                engine.runRenderLoop(() => {
                    scene.render();
                });
                
                updateStatus('renderStatus', 'Running', 'status-success');
                updateStatus('status', 'Ready', 'status-success');

                window.addEventListener("resize", () => {
                    engine.resize();
                });
            } catch (error) {
                showError(error.message || String(error));
                console.error('Initialization error:', error);
            }
        })();
    </script>
