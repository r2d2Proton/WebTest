            <option value="">Loading files...</option>
        </select>
    </div>
    
    <div id="statusPanel">
        <div class="status-line"><span class="status-label">Status:</span> <span id="status" class="status-loading">Initializing...</span></div>
        <div class="status-line"><span class="status-label">Scene:</span> <span id="sceneStatus">-</span></div>
        <div class="status-line"><span class="status-label">Lottie Library:</span> <span id="lottieLibStatus" class="status-loading">Loading...</span></div>
        <div class="status-line"><span class="status-label">Lottie Data:</span> <span id="lottieDataStatus">-</span></div>
        <div class="status-line"><span class="status-label">Texture:</span> <span id="textureStatus">-</span></div>
        <div class="status-line"><span class="status-label">Render Loop:</span> <span id="renderStatus">-</span></div>
        <div class="status-line"><span class="status-label">Animation State:</span> <span id="animationState">-</span></div>
        <div class="status-line" id="errorLine" style="display:none;"><span class="status-label">Error:</span> <span id="errorMsg" class="status-error">-</span></div>
    </div>

    <script>
        let engine;
        let scene;
        let box;
        let mat;
        let lottieTexture;
        let currentLottieAnimation = null;
        let lottieFiles = [];

        window.play = function() {
            try {
                if (lottieTexture) {
                    lottieTexture.lottieAnimation.play();
                    updateStatus('animationState', 'Playing', 'status-success');
                    console.log('Play called');
                } else {
                    console.warn('Lottie texture not available or play method not found');
                }
            } catch (error) {
                console.error('Play error:', error);
            }
        };

        window.pause = function() {
            try {
                if (lottieTexture) {
                    lottieTexture.lottieAnimation.pause();
                    updateStatus('animationState', 'Paused', 'status-loading');
                    console.log('Pause called');
                } else {
                    console.warn('Lottie texture not available or pause method not found');
                }
            } catch (error) {
                console.error('Pause error:', error);
            }
        };

        window.stop = function() {
            try {
                if (lottieTexture) {
                    lottieTexture.lottieAnimation.stop();
                    updateStatus('animationState', 'Stopped', 'status-error');
                    console.log('Stop called');
                } else {
                    console.warn('Lottie texture not available or stop method not found');
                }
            } catch (error) {
                console.error('Stop error:', error);
            }
        };

        function updateStatus(id, message, className = '') {
            const element = document.getElementById(id);
            if (element) {
                element.textContent = message;
                if (className) {
                    element.className = className;
                }
            }
        }

        function showError(message) {
            document.getElementById('errorLine').style.display = 'block';
            updateStatus('errorMsg', message, 'status-error');
            updateStatus('status', 'Error', 'status-error');
        }

        async function fetchLottieFiles() {
            try {
                updateStatus('status', 'Fetching file list...', 'status-loading');
                
                // Fetch the GitHub API to get files in the directory
                const response = await fetch('https://api.github.com/repos/r2d2Proton/LottieSamples/contents/Assets/LottieFiles');
                const files = await response.json();
                
                lottieFiles = files
                    .filter(f => f.name.endsWith('.json'))
                    .map(f => ({
                        name: f.name.replace('.json', '').replace(/_/g, ' '),
                        url: f.download_url
                    }));

                // Populate the select dropdown
                const select = document.getElementById('lottieSelect');
                select.innerHTML = '';
                lottieFiles.forEach((file, index) => {
                    const option = document.createElement('option');
                    option.value = index;
                    option.textContent = file.name;
                    select.appendChild(option);
                });

                if (lottieFiles.length > 0) {
                    select.selectedIndex = 0;
                    updateStatus('status', 'Ready to load', 'status-loading');
                }
            } catch (error) {
                showError('Failed to fetch Lottie files: ' + error.message);
            }
        }

        async function loadLottieAnimation(fileIndex) {
            try {
                if (fileIndex < 0 || fileIndex >= lottieFiles.length) return;

                const file = lottieFiles[fileIndex];
                updateStatus('lottieDataStatus', `Loading ${file.name}...`, 'status-loading');

                // Fetch Lottie JSON data
                const response = await fetch(file.url);
                const lottieObject = await response.json();
                updateStatus('lottieDataStatus', 'Loaded', 'status-success');

                // Remove old texture if exists
                if (lottieTexture) {
                    lottieTexture.dispose();
                }

                // Create new Lottie texture
                updateStatus('textureStatus', 'Creating texture...', 'status-loading');
                lottieTexture = LOTTIE.LottieTexture.ParseFromObject(file.name, lottieObject, scene);
                currentLottieAnimation = lottieTexture;
                mat.albedoTexture = lottieTexture;
                mat.unlit = true;
                updateStatus('textureStatus', 'Applied', 'status-success');
                updateStatus('animationState', 'Loaded', 'status-success');

                updateStatus('status', 'Ready', 'status-success');
            } catch (error) {
                showError('Error loading animation: ' + error.message);
            }
        }

        (async function() {
            try {
                updateStatus('status', 'Creating scene', 'status-loading');
                
                const canvas = document.getElementById("renderCanvas");
                engine = new BABYLON.Engine(canvas, true);
                scene = new BABYLON.Scene(engine);
                updateStatus('sceneStatus', 'Ready', 'status-success');

                const camera = new BABYLON.ArcRotateCamera("camera1", 1, 0.8, 15, new BABYLON.Vector3(0, 0, 0), scene);
                camera.attachControl(canvas, true);

                const light = new BABYLON.HemisphericLight("light1", new BABYLON.Vector3(0, 1, 0), scene);
                light.intensity = 0.7;

                box = BABYLON.MeshBuilder.CreateBox("box", {size: 2}, scene);

                // Load babylonjs-lottie library
                updateStatus('lottieLibStatus', 'Loading library...', 'status-loading');
                await BABYLON.Tools.LoadScriptAsync("https://www.unpkg.com/babylonjs-lottie@1.2.6/dist/index.umd.js");
                updateStatus('lottieLibStatus', 'Loaded', 'status-success');

                // Create material
                mat = new BABYLON.PBRMaterial("lottieMat", scene);
                box.material = mat;

                // Fetch available Lottie files
                await fetchLottieFiles();

                // Load first animation
                if (lottieFiles.length > 0) {
                    await loadLottieAnimation(0);
                }

                // Handle dropdown change
                document.getElementById('lottieSelect').addEventListener('change', (e) => {
                    loadLottieAnimation(parseInt(e.target.value));
                });

                updateStatus('renderStatus', 'Starting...', 'status-loading');
                
                engine.runRenderLoop(() => {
                    scene.render();
                });
                
                updateStatus('renderStatus', 'Running', 'status-success');
                updateStatus('status', 'Ready', 'status-success');

                window.addEventListener("resize", () => {
                    engine.resize();
                });
            } catch (error) {
                showError(error.message || String(error));
            }
        })();
    </script>
